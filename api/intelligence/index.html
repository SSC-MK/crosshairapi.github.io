    Intelligence Endpoints¶

    The Intelligence API provides a comprehensive suite of endpoints for security research, threat intelligence, domain reconnaissance, and vulnerability assessment. These endpoints combine local analysis capabilities with integrations to external intelligence platforms.

    Overview¶

    Category Endpoints API Keys Required
    Domain Intelligence DNS, WHOIS, Domain Info, Subdomains, Tech Detection No
    Security Analysis Security Headers, TLS/SSL Audit No
    Threat Intelligence Multi-source Threat Check, IP Reputation Optional (enhanced with keys)
    Social & Entity Social Profile Search, Entity Graph Extraction No
    TLS Fingerprinting JARM, JA3S, JA4, JA4S, Lookup No
    Censys Host Search, Host Details, Certificate Search Yes
    Vulners CVE Search, Exploit Search, Product Vulns Yes
    SSL Labs SSL Grade, Full Analysis No
    abuse.ch URLhaus, Malware Bazaar, ThreatFox No
    GreyNoise IP Noise/RIOT Check No (Community API)
    AlienVault OTX IP/Domain/Hash Indicators, Pulses Yes

    Base URL

    All endpoints use the base URL http://localhost:8001.

    Breach Intelligence¶

    GET /v1/intel/breach¶

    Search for data breach notifications across state AG offices, SEC filings, and news sources.

    Best For

    Researching security incidents, compliance monitoring, third-party risk assessment, and cyber insurance underwriting.

    Parameters:

    Name Type Description
    company string Company name
    year integer Year of breach
    state string US state code (CA, NY, etc.)
    breach_type string Type: ransomware, phishing, etc.
    industry string Industry: healthcare, finance, etc.
    include_ransomware boolean Include ransomware sources
    limit integer Maximum results (default: 30)
    cursor string Pagination cursor
    fields string Fields to return 
    curl "http://localhost:8001/v1/intel/breach?company=Acme&year=2024&industry=healthcare"

    import requests

def assess_vendor_breaches(vendor_name):
    response = requests.get('http://localhost:8001/v1/intel/breach', params={
        'company': vendor_name,
        'limit': 50
    })
    data = response.json()['data']

    return {
        'vendor': vendor_name,
        'breach_count': data['total'],
        'incidents': [
            {
                'title': r['title'],
                'url': r['url'],
                'date': r.get('published_date')
            }
            for r in data['results']
        ]
    }

    GET /v1/intel/whois¶

    Get WHOIS information for a domain, parsed into structured JSON.

    Best For

    Domain ownership research, expiration monitoring, and brand protection.

    Parameters:

    Name Type Description
    domain string Domain to lookup (required) 
    curl "http://localhost:8001/v1/intel/whois?domain=google.com"

    import requests
from datetime import datetime, timedelta

def check_domain_expiration(domain):
    response = requests.get('http://localhost:8001/v1/intel/whois', params={'domain': domain})
    data = response.json()['data']

    if not data['success']:
        return {'domain': domain, 'error': data.get('error')}

    expiration = data['parsed'].get('expiration_date')
    if expiration:
        exp_date = datetime.fromisoformat(expiration.replace('Z', '+00:00'))
        days_until = (exp_date - datetime.now(exp_date.tzinfo)).days
        return {
            'domain': domain,
            'expires': expiration,
            'days_until': days_until,
            'needs_renewal': days_until < 30
        }
    return {'domain': domain, 'expiration': 'unknown'}

    GET /v1/intel/subdomains¶

    Enumerate subdomains using search engines (passive enumeration -- no direct scanning of target).

    Best For

    Attack surface discovery, finding forgotten/orphaned subdomains, and bug bounty reconnaissance.

    Parameters:

    Name Type Default Description
    domain string - Domain to enumerate (required)
    limit integer 50 Maximum results (1-200)
    cursor string - Pagination cursor 
    curl "http://localhost:8001/v1/intel/subdomains?domain=google.com&limit=10"

    import requests

def enumerate_subdomains(domain):
    response = requests.get('http://localhost:8001/v1/intel/subdomains', params={
        'domain': domain,
        'limit': 200
    })
    data = response.json()['data']

    categories = {
        'api': [], 'dev': [], 'admin': [], 'mail': [], 'other': []
    }

    for sub in data['subdomains']:
        if any(x in sub for x in ['api', 'rest', 'graphql']):
            categories['api'].append(sub)
        elif any(x in sub for x in ['dev', 'staging', 'test', 'beta']):
            categories['dev'].append(sub)
        elif any(x in sub for x in ['admin', 'portal', 'dashboard']):
            categories['admin'].append(sub)
        elif any(x in sub for x in ['mail', 'smtp', 'imap']):
            categories['mail'].append(sub)
        else:
            categories['other'].append(sub)

    return {'domain': domain, 'total': data['total'], 'categories': categories}

    GET /v1/intel/social¶

    Search for a username across social media platforms.

    Best For

    OSINT investigations, background research, brand impersonation detection, and identity correlation.

    Parameters:

    Name Type Description
    username string Username to search (required)
    platforms string Comma-separated: twitter, linkedin, github, instagram, facebook, reddit, youtube
    limit integer Maximum results (default: 20)
    cursor string Pagination cursor
    fields string Fields to return 
    curl "http://localhost:8001/v1/intel/social?username=johndoe&platforms=github,twitter"

    import requests

def find_social_profiles(username, platforms=None):
    params = {'username': username, 'limit': 50}
    if platforms:
        params['platforms'] = ','.join(platforms)

    response = requests.get('http://localhost:8001/v1/intel/social', params=params)
    data = response.json()['data']

    profiles = {}
    for result in data['results']:
        domain = result['domain']
        if domain not in profiles:
            profiles[domain] = []
        profiles[domain].append({
            'title': result['title'],
            'url': result['url']
        })
    return profiles

    GET /v1/intel/entities¶

    Extract entity relationship graph from search results using NLP.

    Best For

    Discovering connections between people, organizations, and places. Ready for visualization with D3.js, vis.js, or Cytoscape.

    Parameters:

    Name Type Required Default Description
    q string Yes - Search query for entity extraction
    limit integer No 30 Number of search results to analyze (5-100)
    max_entities integer No 50 Maximum entities in graph (10-100) 
    curl "http://localhost:8001/v1/intel/entities?q=Tesla+company&limit=20"

    import requests
import networkx as nx

def build_knowledge_graph(query):
    response = requests.get('http://localhost:8001/v1/intel/entities', params={
        'q': query,
        'limit': 50,
        'max_entities': 50
    })
    graph_data = response.json()['data']['graph']

    G = nx.Graph()
    for node in graph_data['nodes']:
        G.add_node(node['id'], label=node['label'],
                   count=node['count'], description=node['description'])
    for edge in graph_data['edges']:
        G.add_edge(edge['source'], edge['target'], weight=edge['weight'])
    return G

    Technology Detection¶

    GET /v1/intel/tech¶

    Detect technologies used by a website. No external APIs required -- all detection is done locally with 50+ technology signatures.

    Best For

    Fingerprinting targets for vulnerability assessment, competitive intelligence, and lead qualification.

    Parameters:

    Name Type Description
    url string URL to analyze (required) 
    curl "http://localhost:8001/v1/intel/tech?url=https://wordpress.org"

    import requests

def analyze_tech_stack(url):
    response = requests.get('http://localhost:8001/v1/intel/tech', params={'url': url})
    data = response.json()['data']

    by_category = {}
    for tech in data['technologies']:
        cat = tech['category']
        if cat not in by_category:
            by_category[cat] = []
        by_category[cat].append({
            'name': tech['name'],
            'confidence': tech['confidence'],
            'version': tech.get('version')
        })
    return {'url': data['url'], 'stack': by_category, 'implied': data['implied']}

    Security Analysis¶

    GET /v1/intel/security-headers¶

    Analyze HTTP security headers and get a security score (0-100) with a letter grade (A+ to F).

    Best For

    Auditing website security configuration, validating security header deployment, and compliance verification.

    Parameters:

    Name Type Description
    url string URL to analyze (required) 
    curl "http://localhost:8001/v1/intel/security-headers?url=https://github.com"

    import requests

def audit_security_headers(urls):
    results = []
    for url in urls:
        response = requests.get('http://localhost:8001/v1/intel/security-headers', params={'url': url})
        data = response.json()['data']
        results.append({

            'score': data['score'],
            'grade': data['grade'],
            'missing': len(data['missing_headers']),

    import requests

def check_cert_expiration(domains):
    alerts = []
    for domain in domains:
        response = requests.get('http://localhost:8001/v1/intel/tls', params={'domain': domain})
        data = response.json()['data']
        cert = data['certificate']

        if cert['is_expired']:
            alerts.append({'domain': domain, 'status': 'EXPIRED', 'days': cert['days_until_expiry']})
        elif cert['days_until_expiry'] < 30:
            alerts.append({'domain': domain, 'status': 'CRITICAL', 'days': cert['days_until_expiry']})
        elif cert['days_until_expiry'] < 90:
            alerts.append({'domain': domain, 'status': 'WARNING', 'days': cert['days_until_expiry']})
    return alerts

    import requests

def analyze_suspicious_ips(ip_list):
    results = []
    for ip in ip_list:
        response = requests.get('http://localhost:8001/v1/intel/ip', params={'ip': ip})
        data = response.json()['data']

        risk_score = 0
        risk_factors = []

        if data.get('abuse') and data['abuse'].get('confidence_score', 0) > 50:
            risk_score += 30
            risk_factors.append(f"High abuse score: {data['abuse']['confidence_score']}%")
        if data.get('abuse', {}).get('is_tor'):
            risk_score += 20
            risk_factors.append("Tor exit node")
        if data.get('shodan', {}).get('vulns'):
            risk_score += 25
            risk_factors.append(f"Known vulnerabilities: {len(data['shodan']['vulns'])}")

        results.append({
            'ip': ip, 'risk_score': risk_score,
            'risk_factors': risk_factors,
            'org': data.get('org'),
            'country': data.get('geolocation', {}).get('country_code')
        })
    return sorted(results, key=lambda x: x['risk_score'], reverse=True)
